from Crypto.Cipher import AES import base64 import os # Define the encryption key (must be 16, 24, or 32 bytes long) key = b'ThisIsASecretKey123' # 16-byte key (AES-128) # Function to pad data to be a multiple of AES block size def pad(data): block_size = AES.block_size # AES block size is 16 bytes padding_length = block_size - (len(data) % block_size) padding = chr(padding_length) * padding_length return data + padding # Encrypt the HTML file content html_content = """ Project 1

AM Dataset for GenAI Models

This section provides a collection of diverse AM datasets...

Dataset Statistics

AM Phase MCQs True/False Free-Form
Design 6 12 2
""" # Convert HTML content to bytes data_bytes = pad(html_content).encode() # Generate a random initialization vector (IV) iv = os.urandom(16) # Encrypt data using AES-CBC mode cipher = AES.new(key, AES.MODE_CBC, iv) encrypted_data = cipher.encrypt(data_bytes) # Encode in base64 for safe storage and transmission encrypted_html = base64.b64encode(iv + encrypted_data).decode() encrypted_html
